#include <stdio.h>
 
void print(char*);
 
int main() {
   char s[100];
   gets(s);
   print(s);
   return 0;
}
 
void print(char *t) {
   if (*t == '\0')
      return;
   printf("%c", *t);
   print(++t);
}

#include <stdio.h>
#include <string.h>
 
int main() {
   char s[100];
   int c, l;
 
   gets(s);
 
   l = strlen(s);
 
   for (c = 0; c < l; c++)
      printf("%c", s[c]);
 
   return 0;
}
